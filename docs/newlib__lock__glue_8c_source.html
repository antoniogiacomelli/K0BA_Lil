351int __cxa_guard_acquire(CxaGuardObject_t *guard_object)
352{
353 STM32_LOCK_BLOCK_IF_NULL_ARGUMENT(guard_object);
354
355 if (atomic_load(&guard_object->initialized) == 0)
356 {
357 /* Object needs initialization, lock threading context */
358 stm32_lock_acquire(&__cxa_guard_mutex);
359 if (atomic_load(&guard_object->initialized) == 0)
360 {
361 /* Object needs initialization */
362 if (guard_object->acquired)
363 {
364 /* Object initialization already in progress */
365 STM32_LOCK_BLOCK();
366 }
367
368 /* Lock acquired */
369 guard_object->acquired = 1;
370 return 1;
371 }
372 else
373 {
374 /* Object initialized in another thread */
375 stm32_lock_release(&__cxa_guard_mutex);
376 }
377 }
378
379 /* Object already initialized */
380 return 0;
381}

387void __cxa_guard_abort(CxaGuardObject_t *guard_object)
388{
389 STM32_LOCK_BLOCK_IF_NULL_ARGUMENT(guard_object);
390
391 if (guard_object->acquired)
392 {
393 /* Release lock */
394 guard_object->acquired = 0;
395 stm32_lock_release(&__cxa_guard_mutex);
396 }
397 else
398 {
399 /* Trying to release non-acquired lock */
400 STM32_LOCK_BLOCK();
401 }
402}

410 STM32_LOCK_BLOCK_IF_NULL_ARGUMENT(guard_object);
