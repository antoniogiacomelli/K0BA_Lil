105K_TCB* kTCBQSearchPID(K_TCBQ* const self, PID uPid)
106{
107 if (self == NULL || self->listDummy.nextPtr == &(self->listDummy))
108 {
109 return NULL;
110 }
111 else
112 {
113 K_LISTNODE* currNodePtr = K_LIST_HEAD(self);
114 while(currNodePtr != &(self->listDummy))
115 {
116 K_TCB* currTcbPtr = K_GET_TCB_ADDR(currNodePtr, K_TCB);
117 if (currTcbPtr->uPid == uPid)
118 {
119 return currTcbPtr;
120 }
121 currNodePtr = currNodePtr->nextPtr;
122 }
123 }
124 return NULL;
125}

128// Enqueue task to the ready queue
129K_ERR kReadyQEnq(K_TCB* const tcbPtr)
130{
131 K_CR_AREA;
132 K_ENTER_CR;
133 if (IS_NULL_PTR(tcbPtr))
134 {
135 kErrHandler(FAULT_NULL_OBJ);
136 K_EXIT_CR;
137 return K_ERR_NULL_OBJ;
138 }
139 if (tcbPtr->realPrio != tcbPtr->priority)
140 assert(0);
141 if (kTCBQEnq(&readyQueue[tcbPtr->priority], tcbPtr) == K_SUCCESS)
142 {
143 tcbPtr->status = READY;
144 tcbPtr->pendingObj=NULL;
145#ifdef INSTANT_PREEMPT_LOWER_PRIO
146 if (READY_HIGHER_PRIO(tcbPtr))
147 {
148 K_PEND_CTXTSWTCH;
149 K_EXIT_CR;
150 }
151#endif
152 return K_SUCCESS;
153 }
154 K_EXIT_CR;
155 return K_ERROR;
156}
