72K_ERR kMailboxInit(K_MAILBOX* const self)
73{
74 if (IS_NULL_PTR(self))

77 K_ENTER_CR;

93K_ERR kMailboxPost(K_MAILBOX* self, const ADDR mesgPtr, SIZE mesgSize)
94{
95 if (IS_NULL_PTR(self) || IS_NULL_PTR(mesgPtr))
96 kErrHandler(FAULT_NULL_OBJ);
97
98 assert(!kSemaWait(&self->semaEmpty)); /*D EMPTY*/
99 assert(!kMutexLock(&self->mailMutex));
100 K_CR_AREA;
101 K_ENTER_CR;
102 kMemCpy((ADDR)&self->mail.mesgPtr, mesgPtr, mesgSize);
103 self->mail.senderPid = runPtr->uPid;
104 self->mail.mesgSize = mesgSize;
105 K_EXIT_CR;
106 kMutexUnlock(&self->mailMutex);
107 kSemaSignal(&self->semaFull);
108 kSemaWait(&self->semaAck);
109 /*D ACK*/
110 return K_SUCCESS;
111}

151K_ERR kMesgQPut(K_MESGQ* const self, ADDR mesgPtr, SIZE mesgSize)
152{
153 if (IS_NULL_PTR(self) || IS_NULL_PTR(mesgPtr))
154 kErrHandler(FAULT_NULL_OBJ);
155 kSemaWait(&self->semaSpace);
156 kMutexLock(&(self->mesgQMutex));
157 K_CR_AREA;
158 K_ENTER_CR;
159 K_MESGBUFF* mesgBuffPtr = kMesgBuffGet();
160 mesgBuffPtr->mesgPtr = (ADDR)kMemAlloc(&self->mesgMemCtrlBlk);
161 if (mesgBuffPtr->mesgPtr == NULL)
162 assert(0);
163 kMemCpy((ADDR)mesgBuffPtr->mesgPtr, mesgPtr, mesgSize);
164 mesgBuffPtr->senderPid = runPtr->uPid;
165 mesgBuffPtr->mesgSize = mesgSize;
166 K_EXIT_CR;
167 kListAddTail(&(self->mesgList), &mesgBuffPtr->mesgNode);
168 kMutexUnlock(&(self->mesgQMutex));
169 kSemaSignal(&(self->semaNewMesg));

189 kMutexUnlock(&(self->mesgQMutex));

204 if (IS_NULL_PTR(self))

206 self->head=0;

215SIZE kPipeRead(K_PIPE* const self, BYTE* destPtr, SIZE nBytes)
216{
217 if (IS_NULL_PTR(self))
218 kErrHandler(FAULT_NULL_OBJ);
219 SIZE readBytes = 0;
220 if (nBytes == 0)
221 return 0;
222
223 while (nBytes) /* while there still bytes to be read */
224 {
225 kMutexLock(&(self->mutex));
226 while(self->data) /* while there is data */
227 {
228 *destPtr++ = self->buffer[self->tail++]; /* read from the tail */
229 self->tail %= PIPE_SIZE; /* wrap around */
230 self->data--; /* decrease data */
231 self->room++; /* increase room */
232 readBytes++; /* increase read bytes */
233 nBytes--; /* decrease asked bytes number */
234
235 if (nBytes==0) /* read all bytes asked */
236 break; /* break reading loop */
237 }
238 kMutexUnlock(&(self->mutex));
239 kCondWake(&(self->condRoom)); /*either there is no more data or all req data was read*/
240
241 if (readBytes > 0)
242 {
243 return readBytes; /* return number of read bytes */
244 }
245 /* if here, no data was read */
246 kCondWait(&(self->condData)); /* wait for data from writers */
247 }
248 return 0;
249}

251SIZE kPipeWrite(K_PIPE* const self, const BYTE* srcPtr, SIZE nBytes)
252{
253 if (IS_NULL_PTR(self))
254 kErrHandler(FAULT_NULL_OBJ);
255 SIZE writeBytes = 0;
256 if (nBytes == 0)
257 return 0;
258 if (nBytes <= 0)
259 return 0;
260 while (nBytes) /* while there still bytes to be written */
261 {
262 kMutexLock(&(self->mutex));
263 while(self->room) /* while there is room */
264 {
265 self->buffer[self->head++] = *srcPtr;
266 self->head %= PIPE_SIZE;
267 self->data++;
268 self->room--;
269 writeBytes++;
270 srcPtr++;
271 nBytes--;
272 if (nBytes==0)
273 {
274 break;
275 }
276 }
277 kMutexUnlock(&(self->mutex));
278 if (writeBytes)
279 {
280 kCondWake(&(self->condData));
281 }
282 if (nBytes==0) /* finished writing */
283 {
284 return writeBytes;
285 }
286 kCondWait(&(self->condRoom)); /*still data to write, but no room*/
287 }
288 return 0;
289}
