89K_ERR kMailboxPost(K_MAILBOX* self, const ADDR msgPtr, SIZE msgSize)
90{
91 if (IS_NULL_PTR(self) || IS_NULL_PTR(msgPtr))
92 kErrHandler(FAULT_NULL_OBJ);
93
94 assert(!kSemaWait(&self->semaEmpty)); /*D EMPTY*/
95 assert(!kMutexLock(&self->mailMutex));
96 K_CR_AREA;
97 K_ENTER_CR;
98 kMemCpy((ADDR)&self->mail.msgPtr, msgPtr, msgSize);
99 self->mail.senderPid = runPtr->uPid;
100 self->mail.msgSize = msgSize;
101 K_EXIT_CR;
102 kMutexUnlock(&self->mailMutex);
103 kSemaSignal(&self->semaFull);
104 kSemaWait(&self->semaAck);
105 /*D ACK*/
106 return K_SUCCESS;
107}

147K_ERR kMsgQueueSend(K_MSGQ* const self, ADDR msgPtr, SIZE msgSize)
148{
149 if (IS_NULL_PTR(self) || IS_NULL_PTR(msgPtr))
150 kErrHandler(FAULT_NULL_OBJ);
151 kSemaWait(&self->semaSpace);
152 kMutexLock(&(self->msgQMutex));
153 K_CR_AREA;
154 K_ENTER_CR;
155 K_MSGBUFF* msgBuffPtr = kMsgBuffGet();
156 msgBuffPtr->msgPtr = (ADDR)kMemAlloc(&self->msgMemCtrlBlk);
157 if (msgBuffPtr->msgPtr == NULL)
158 assert(0);
159 kMemCpy((ADDR)msgBuffPtr->msgPtr, msgPtr, msgSize);
160 msgBuffPtr->senderPid = runPtr->uPid;
161 msgBuffPtr->msgSize = msgSize;
162 K_EXIT_CR;
163 kListAddTail(&(self->msgList), &msgBuffPtr->msgNode);
164 kMutexUnlock(&(self->msgQMutex));
165 kSemaSignal(&(self->semaNewMsg));
166 return K_SUCCESS;
167}

184 K_EXIT_CR;
185 kMutexUnlock(&(self->msgQMutex));
186 kSemaSignal(&self->semaSpace);

211SIZE kPipeRead(K_PIPE* const self, BYTE* destPtr, SIZE nBytes)
212{
213 if (IS_NULL_PTR(self))
214 kErrHandler(FAULT_NULL_OBJ);
215 SIZE readBytes = 0;
216 if (nBytes == 0)
217 return 0;
218
219 while (nBytes) /* while there still bytes to be read */
220 {
221 kMutexLock(&(self->mutex));
222 while(self->data) /* while there is data */
223 {
224 *destPtr++ = self->buffer[self->tail++]; /* read from the tail */
225 self->tail %= PIPE_SIZE; /* wrap around */
226 self->data--; /* decrease data */
227 self->room++; /* increase room */
228 readBytes++; /* increase read bytes */
229 nBytes--; /* decrease asked bytes number */
230
231 if (nBytes==0) /* read all bytes asked */
232 break; /* break reading loop */
233 }
234 kMutexUnlock(&(self->mutex));
235 kCondVarSignalBroadcast(&(self->condRoom)); /*either there is no more data or all req data was read*/
236
237 if (readBytes > 0)
238 {
239 return readBytes; /* return number of read bytes */
240 }
241 /* if here, no data was read */
242 kCondVarWait(&(self->condData)); /* wait for data from writers */
243 }
244 return 0;
245}

247SIZE kPipeWrite(K_PIPE* const self, const BYTE* srcPtr, SIZE nBytes)
248{
249 if (IS_NULL_PTR(self))
250 kErrHandler(FAULT_NULL_OBJ);
251 SIZE writeBytes = 0;
252 if (nBytes == 0)
253 return 0;
254 if (nBytes <= 0)
255 return 0;
256 while (nBytes) /* while there still bytes to be written */
257 {
258 kMutexLock(&(self->mutex));
259 while(self->room) /* while there is room */
260 {
261 self->buffer[self->head++] = *srcPtr;
262 self->head %= PIPE_SIZE;
263 self->data++;
264 self->room--;
265 writeBytes++;
266 srcPtr++;
267 nBytes--;
268 if (nBytes==0)
269 {
270 break;
271 }
272 }
273 kMutexUnlock(&(self->mutex));
274 if (writeBytes)
275 {
276 kCondVarSignalBroadcast(&(self->condData));
277 }
278 if (nBytes==0) /* finished writing */
279 {
280 return writeBytes;
281 }
282 kCondVarWait(&(self->condRoom)); /*still data to write, but no room*/
283 }
284 return 0;
285}
